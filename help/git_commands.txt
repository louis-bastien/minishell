###Git Useful Commands###


#Updating the main repository directly

git clone <repository-url>
git add *
git commit -m "descrition of changes"
git push origin main                    //Origin = Github repository, Main = Branch name

#Creating and updating a branch

git checkout -b <branch-name>           //Switch to branch (and create if does not exist)
git add *
git commit -m "descrition of changes"
git push origin <branch-name>           //Update the branch on remote repository (Github)
                                        //Then you can make a pull request on Github to ask to merge the branch into  main

#Merging a branch changes into the Main branch (without pull request)

git checkout -b <branch-name>           //Switch to branch
git add *
git commit -m "descrition of changes"
git checkout main //Switch to main
git merge <branch-name>                 //Apply branch changes to the current branch (main)

#Deleting branch

git branch -d <branch-name>             //Remove branch on local machine
git push origin --delete <branch-name>  //Remove branch on remote repository
